from django.shortcuts import render
from .models import *

def index(request):

def submit(request):
    obj= Todo()
    obj.title= request.GET['title']
    obj.description= request.GET['description']
    obj.priority= request.GET['priority']
    obj.save()
    return render(request,'index.html')
